class Class

class DClass {
  mutable : Boolean = true
}

class Field
class Method
class Parameter

class DAttr {
  mutable  : Boolean = true
  optional : Boolean = true
  length   : Integer = -1
  unique   : Boolean = false
  id       : Boolean = false
  auto     : Boolean = false
  min      : Double = -∞
  max      : Double = +∞
}

class DAssoc {
  ascName   : String
  role      : String
  ascType   : AssocType
  endType   : AssocEndType
  associate : Associate
}

class Associate {
  type    : Class
  cardMin : Integer
  cardMax : Integer
}

class AttrRef {
  value : String
}

class DOpt {
  type     : OptType
  requires : String
  effects  : String
}

enum OptType <<enumeration>> {
  RequiredConstructor
  ObjectFormConstructor
  AutoAttributeValueGen
  LinkCountGetter
  LinkCountSetter
  LinkAdder
  LinkAdderNew
  LinkRemover
  LinkUpdater
  Setter
  Getter
}

enum AssocType <<enumeration>> {
  OneOne
  OneMany
  ManyMany
}

enum AssocEndType <<enumeration>> {
  One
  Many
}
